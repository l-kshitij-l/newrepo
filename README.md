       TO_CHAR(REQUESTED_DATE,'YYYY-MM-DD') AS REQUESTED_DATE,
       COUNT(1) as COUNT
FROM SMS_PUSH_MESSAGE_LOG_CREDIT_AR spm
LEFT JOIN PRM_USERS pu ON spm.NREVID = pu.lid
WHERE spm.CHANNEL_NAME = 'IMPC'
AND spm.REQUESTED_DATE BETWEEN TO_DATE('2026-05-08','YYYY-MM-DD') AND TO_DATE('2026-06-04','YYYY-MM-DD')
GROUP BY RULE_ID, RULE_NAME, pu.sname, REQUESTED_DATE
ORDER BY REQUESTED_DATE DESC;

)
SELECT COUNT(row_num) FROM cte;

create or replace PROCEDURE "USP_CREDIT_SMS_REPORT_ARCHIVAL_INSERTION" 
AS
V_MSG VARCHAR2(2000);
V_START_DATE DATE ;
V_END_DATE DATE := TRUNC(SYSDATE) - 1;
V_INSERT_COUNT NUMBER := 0;

BEGIN
    V_MSG := 'USP_CREDIT_SMS_REPORT_ARCHIVE JOB Started.';
    PP_MAKE_AUDIT_ENTRY(V_MSG,11,59);

    SELECT MAX(REQUESTED_DATE) INTO V_START_DATE FROM SMS_PUSH_MESSAGE_LOG_CREDIT_AR ;
    V_START_DATE := V_START_DATE + 1;

    IF V_START_DATE IS NULL THEN
        SELECT MIN(REQUESTED_DATE) INTO V_START_DATE FROM SMS_PUSH_MESSAGE_LOG_CREDIT_CHANNEL;
    ELSIF V_START_DATE IS NULL THEN
        V_MSG := 'No records found to insert in the SMS_PUSH_MESSAGE_LOG_CREDIT_AR table.';
        PP_MAKE_AUDIT_ENTRY(V_MSG,11,59);
        V_MSG := 'USP_CREDIT_SMS_REPORT_ARCHIVE JOB Completed.';
        PP_MAKE_AUDIT_ENTRY(V_MSG,11,59);
        RETURN;
    END IF;

    WHILE TRUNC(V_START_DATE) <= TRUNC(V_END_DATE) LOOP
        BEGIN
           MERGE INTO SMS_PUSH_MESSAGE_LOG_CREDIT_AR A
        USING (SELECT * FROM SMS_PUSH_MESSAGE_LOG_CREDIT_CHANNEL
        WHERE REQUESTED_DATE = TRUNC(V_START_DATE)) U
        ON (A.MESSAGE_ID = U.MESSAGE_ID
        AND A.ACTION_LEVEL_KEY=U.ACTION_LEVEL_KEY 
        AND A.ACTION_LEVEL_ID=U.ACTION_LEVEL_ID 
        AND A.ACTION_TYPE_ID=U.ACTION_TYPE_ID
       )

        WHEN MATCHED THEN
       --  update records if they exist
        UPDATE SET  A.ACCT_NUMBER      = U.ACCT_NUMBER,
                    A.MESSAGETEXT      = U.MESSAGETEXT,
                    A.TEMPLATE_ID      = U.TEMPLATE_ID,
                    A.LSTATE1          = U.LSTATE1,
                    A.LSTATUS1         = U.LSTATUS1,
                    A.DTREQUESTED1     = U.DTREQUESTED1,
                    A.ISAUTOSMS        = U.ISAUTOSMS,
                    A.TDDATEKEY        = U.TDDATEKEY,
                    A.NREVID           = U.NREVID,
                    A.LPORTFOLIOID     = U.LPORTFOLIOID,
                    A.CHANNEL_ID       = U.CHANNEL_ID,
                    A.CHANNEL_NAME     = U.CHANNEL_NAME,
                    A.REFNO            = U.REFNO,
                    A.SD_TIEBREAKER    = U.SD_TIEBREAKER,
                    A.RETRY_ATTEMPTS   = U.RETRY_ATTEMPTS,
                    A.DD_PARTITION_DATE = U.DD_PARTITION_DATE,
                    A.RULE_ID          = U.RULE_ID,
                    A.RULE_NAME        = U.RULE_NAME,
                    A.REQUESTED_DATE   = U.REQUESTED_DATE,
                    A.FILLER1          = U.FILLER1,
                    A.FILLER2         = 'UPDATED_ENTRY'

       WHEN NOT MATCHED THEN
        -- Insert new records if they do not exist in the destination table
        INSERT (MESSAGE_ID,ACCT_NUMBER,MESSAGETEXT,TEMPLATE_ID,LSTATE1,LSTATUS1,DTREQUESTED1,ISAUTOSMS,ACTION_LEVEL_KEY,ACTION_LEVEL_ID,
        TDDATEKEY,NREVID,ACTION_TYPE_ID,LPORTFOLIOID,CHANNEL_ID,CHANNEL_NAME,REFNO,SD_TIEBREAKER,RETRY_ATTEMPTS,DD_PARTITION_DATE,RULE_ID,RULE_NAME,REQUESTED_DATE,FILLER1,filler2) 
        VALUES(U.MESSAGE_ID,U.ACCT_NUMBER,U.MESSAGETEXT,U.TEMPLATE_ID,U.LSTATE1,U.LSTATUS1,U.DTREQUESTED1,U.ISAUTOSMS,U.ACTION_LEVEL_KEY,U.ACTION_LEVEL_ID,
        U.TDDATEKEY,U.NREVID,U.ACTION_TYPE_ID,U.LPORTFOLIOID,U.CHANNEL_ID,U.CHANNEL_NAME,U.REFNO,U.SD_TIEBREAKER,U.RETRY_ATTEMPTS,U.DD_PARTITION_DATE,U.RULE_ID,U.RULE_NAME,U.REQUESTED_DATE,U.FILLER1,'INSERTED_ENTRY'); 

            V_INSERT_COUNT := V_INSERT_COUNT + SQL%ROWCOUNT;

            IF SQL%ROWCOUNT = 0 THEN
                V_MSG := 'No record found in SMS_PUSH_MESSAGE_LOG_CREDIT_CHANNEL table for date: '|| TO_CHAR(V_START_DATE,'DD-MON-YYYY');
                PP_MAKE_AUDIT_ENTRY(V_MSG,11,59);
            ELSE 
                V_MSG := SQL%ROWCOUNT||' record(s) inserted in SMS_PUSH_MESSAGE_LOG_CREDIT_AR table for date: '||TO_CHAR(V_START_DATE,'DD-MON-YYYY');
                PP_MAKE_AUDIT_ENTRY(V_MSG,11,59);
            END IF;

            V_START_DATE := V_START_DATE + 1; 

            EXCEPTION
                WHEN OTHERS THEN
                ROLLBACK;
                V_START_DATE := V_START_DATE + 1; 
                V_MSG := SUBSTR(DBMS_UTILITY.FORMAT_ERROR_BACKTRACE || ',SQLERRM:' || SQLERRM, 1, 2000);
                PP_MAKE_AUDIT_ENTRY(V_MSG, 11, 59);
                COMMIT; 
            RAISE;
        END;
    END LOOP;

    COMMIT;

    V_MSG := 'USP_CREDIT_SMS_REPORT_ARCHIVE JOB Completed. Total record(s) inserted: ' || V_INSERT_COUNT;
    PP_MAKE_AUDIT_ENTRY (V_MSG,11,59);

EXCEPTION
    WHEN OTHERS THEN
    ROLLBACK;
    V_MSG := SUBSTR(DBMS_UTILITY.FORMAT_ERROR_BACKTRACE || ',SQLERRM:' || SQLERRM, 1, 2000);
    PP_MAKE_AUDIT_ENTRY(V_MSG, 11, 59);
    COMMIT; 
END;
